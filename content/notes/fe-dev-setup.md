```
# create package.json
# npm init

# install specific version of npm package
$ npm install --save react@0.14.7

# browserify
$ npm install -g browserify

# Babel + browserify
$ npm install --save-dev babelify

# build
$ npm run build

# sass
$npm install --save-dev node-sass
```
